228 template<typename T>
229 ignite_result<T> result_of_operation(const std::function<T()> &operation) noexcept {
230  try {
231  if constexpr (std::is_same<decltype(operation()), void>::value) {
232  operation();
233  return {};
234  } else {
235  return {operation()};
236  }
237  } catch (const ignite_error &err) {
238  return {ignite_error(err)};
239  } catch (const std::exception &err) {
240  std::string msg("Standard library exception is thrown: ");
241  msg += err.what();
242  return {ignite_error(error::code::INTERNAL, msg, std::current_exception())};
243  } catch (...) {
244  return {ignite_error(error::code::INTERNAL, "Unknown error is encountered when processing network event",
245  std::current_exception())};
246  }
247 }

255 template<typename T>
256 void result_set_promise(std::promise<T> &pr, ignite_result<T> &&res) {
257  if (!res) {
258  pr.set_exception(std::make_exception_ptr(std::move(res).error()));
259  } else {
260  if constexpr (std::is_same<T, void>::value) {
261  pr.set_value();

263  pr.set_value(std::move(res).value());
