152  template<typename T>
153  void set(std::string_view name, T &&value) {
154  auto parsed_name = parse_name(name);
155  auto it = m_indices.find(parsed_name);
156  if (it != m_indices.end()) {
157  auto idx = it->second;
158  m_pairs[idx].second = std::forward<T>(value);
159  return;
160  }
161 
162  m_pairs.emplace_back(name, std::forward<T>(value));
163  m_indices.emplace(parsed_name, m_pairs.size() - 1);
164  }

212  [[nodiscard]] static std::string parse_name(std::string_view name) {
213  if (name.size() >= 2 && name.front() == '"' && name.back() == '"')
214  name = name.substr(1, name.size() - 2);
215 
216  if (name.empty())
217  throw ignite_error("Column name can not be an empty string");
218 
219  std::string res;

221

224  }
